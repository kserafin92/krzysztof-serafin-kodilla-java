                        ID INT AUTO_INCREMENT PRIMARY KEY,

                        SUMMARY VARCHAR(255) NOT NULL,
                        DESCRIPTION TEXT,
                        USER_ID_ASSIGNEDTO INT,
                        FOREIGN KEY (ISSUESLIST_ID) REFERENCES ISSUESLISTS(ID)
);
